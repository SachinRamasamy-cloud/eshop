<body class="dark-mode">

    <button class="theme-toggle" id="themeToggle">Toggle Theme</button>

    <div>
        <label for="amount">Choose Amount:</label>
        <select id="amount"></select>
        <p id="estimate">Estimated Price: ₹0</p>
    </div>

<script>
    // Example product
    const product = {
        fromprice: 500,
        toPrice: 1000
    };

    const select = document.getElementById('amount');
    const estimateEl = document.getElementById('estimate');

    // Populate dropdown dynamically
    for (let price = product.fromprice; price <= product.toPrice; price += 50) {
        let option = document.createElement('option');
        option.value = price;
        option.textContent = `₹${price}`;
        option.style.backgroundColor = 'var(--cardBg)';
        option.style.color = 'var(--textColor)';
        select.appendChild(option);
    }

    // Update estimated price
    select.addEventListener('change', function () {
        estimateEl.textContent = `Estimated Price: ₹${parseInt(this.value) + 10}`;
    });

    // Trigger default selection
    if (select.options.length > 0) {
        select.dispatchEvent(new Event('change'));
    }

    // Theme toggle
    const themeToggle = document.getElementById('themeToggle');
    themeToggle.addEventListener('click', () => {
        document.body.classList.toggle('dark-mode');
    });
</script>
